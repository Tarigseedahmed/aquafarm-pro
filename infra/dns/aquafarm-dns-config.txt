## A Records (replace YOUR_VPS_IP with actual IP)
# Point main domain and subdomains to VPS IP

# Main domain
aquafarm.cloud.                A       YOUR_VPS_IP

# WWW subdomain
www.aquafarm.cloud.           A       YOUR_VPS_IP

# API subdomain
api.aquafarm.cloud.           A       YOUR_VPS_IP

# Admin panel
admin.aquafarm.cloud.         A       YOUR_VPS_IP

# Development/staging (optional)
dev.aquafarm.cloud.           A       YOUR_VPS_IP
staging.aquafarm.cloud.       A       YOUR_VPS_IP

## CNAME Records (alternative to A records for subdomains)
# You can use CNAME instead of A records for subdomains:
# www.aquafarm.cloud.           CNAME   aquafarm.cloud.
# api.aquafarm.cloud.           CNAME   aquafarm.cloud.
# admin.aquafarm.cloud.         CNAME   aquafarm.cloud.

## MX Records (Email)
# For email functionality (using Hostinger email or external provider)
aquafarm.cloud.               MX  10  mail.aquafarm.cloud.
aquafarm.cloud.               MX  20  mail2.hostinger.com.

# Mail server A record
mail.aquafarm.cloud.          A       YOUR_VPS_IP

## TXT Records

# SPF Record (Email security)
aquafarm.cloud.               TXT     "v=spf1 include:_spf.hostinger.com ~all"

# DKIM Record (Email authentication)
# This will be provided by your email service
default._domainkey.aquafarm.cloud.    TXT     "v=DKIM1; k=rsa; p=YOUR_DKIM_PUBLIC_KEY"

# DMARC Record (Email policy)
_dmarc.aquafarm.cloud.        TXT     "v=DMARC1; p=quarantine; rua=mailto:dmarc@aquafarm.cloud"

# Domain verification (for SSL certificates)
aquafarm.cloud.               TXT     "v=spf1 include:_spf.hostinger.com ~all"

## SRV Records (if using special services)
# Example for WebSocket or special protocols
# _service._protocol.aquafarm.cloud.    SRV     0 5 PORT aquafarm.cloud.

## CAA Records (SSL Certificate Authority Authorization)
# Allow Let's Encrypt and other CAs to issue certificates
aquafarm.cloud.               CAA     0 issue "letsencrypt.org"
aquafarm.cloud.               CAA     0 issue "digicert.com"
aquafarm.cloud.               CAA     0 iodef "mailto:ssl-admin@aquafarm.cloud"
